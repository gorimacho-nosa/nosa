async/awaitとは？

async/awaitはPromiseをより直感的に扱うための構文糖衣（シンタックスシュガー）です。

非同期処理を同期処理のように書けるため、コードがより読みやすくなります。

🍳 日常生活での例え

料理のレシピを書くときの違いのようなものです！

Promiseとasync/awaitの比較

Promiseチェーン

function fetchUserData() {
    return fetch('/api/user')
        .then(response => response.json())
        .then(user => {
            return fetch(`/api/posts/${user.id}`);
        })
        .then(response => response.json())
        .then(posts => {
            console.log(posts);
        })
        .catch(error => {
            console.error(error);
        });
}

async/await

async function fetchUserData() {
    try {
        const userResponse = await fetch('/api/user');
        const user = await userResponse.json();
        
        const postsResponse = await fetch(`/api/posts/${user.id}`);
        const posts = await postsResponse.json();
        
        console.log(posts);
    } catch (error) {
        console.error(error);
    }
}

基本的な構文

1. async関数の定義

関数の前にasyncキーワードを付ける

// 関数宣言
async function myFunction() {
    // ...
}

// 関数式
const myFunction = async () => {
    // ...
}

// メソッド

    async myMethod() {
        // ...
    }
}

2. awaitの使用

Promiseの前にawaitキーワードを付ける

async function example() {
    // awaitはasync関数内でのみ使用可能
    const result = await somePromise();
    console.log(result);
}

基本的な例

// 遅延を作る関数
function delay(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
}

// async/awaitを使った関数
async function greetWithDelay() {
    console.log('こんにちは！');
    
    await delay(2000); // 2秒待つ
    console.log('2秒経ちました');
    
    await delay(1000); // 1秒待つ
    console.log('さらに1秒経ちました');
    
    return '挨拶完了！';
}

// 実行
greetWithDelay().then(result => {
    console.log(result);
});

エラー処理

try-catchを使ったエラー処理

async function fetchDataWithErrorHandling() {
    try {
        // 成功する処理
        const data1 = await Promise.resolve('データ1取得成功');
        console.log(data1);
        
        // 失敗する処理
        const data2 = await Promise.reject(new Error('データ2取得失敗'));
        console.log(data2); // ここは実行されない
        
    } catch (error) {
        console.error('エラーをキャッチ:', error.message);
    } finally {
        console.log('処理終了');
    }
}

fetchDataWithErrorHandling();

複数のエラーハンドリング

async function robustFetch(url) {
    const maxRetries = 3;
    
    for (let i = 0; i < maxRetries; i++) {
        try {
            const response = await fetch(url);
            if (!response.ok) {
                throw new Error(`HTTP error! status: ${response.status}`);
            }
            return await response.json();
            
        } catch (error) {
            console.log(`試行 ${i + 1} 失敗:`, error.message);
            
            if (i === maxRetries - 1) {
                throw error; // 最後の試行でも失敗したらエラーを投げる
            }
            
            // 次の試行まで少し待つ
            await new Promise(resolve => setTimeout(resolve, 1000));
        }
    }
}

並列処理

逐次実行 vs 並列実行

// APIを模擬する関数
async function fetchData(name, delay) {
    await new Promise(resolve => setTimeout(resolve, delay));
    return `${name}のデータ`;
}

// ❌ 逐次実行（遅い）
async function sequentialFetch() {
    console.time('逐次実行');
    
    const data1 = await fetchData('API1', 1000); // 1秒待つ
    const data2 = await fetchData('API2', 1000); // さらに1秒待つ
    const data3 = await fetchData('API3', 1000); // さらに1秒待つ
    
    console.log([data1, data2, data3]);
    console.timeEnd('逐次実行'); // 約3秒
}

// ✅ 並列実行（速い）
async function parallelFetch() {
    console.time('並列実行');
    
    const [data1, data2, data3] = await Promise.all([
        fetchData('API1', 1000),
        fetchData('API2', 1000),
        fetchData('API3', 1000)
    ]);
    
    console.log([data1, data2, data3]);
    console.timeEnd('並列実行'); // 約1秒
}

実践的な例

実用例1: APIチェーン

// ユーザー情報と投稿を取得する実践的な例
async function getUserWithPosts(userId) {
    try {
        // ユーザー情報を取得
        const user = await fetch(`/api/users/${userId}`).then(r => r.json());
        
        // ユーザーの投稿を並列で取得
        const [posts, profile] = await Promise.all([
            fetch(`/api/users/${userId}/posts`).then(r => r.json()),
            fetch(`/api/users/${userId}/profile`).then(r => r.json())
        ]);
        
        // データを整形
        return {
            ...user,
            profile,
            posts,
            postCount: posts.length
        };
        
    } catch (error) {
        console.error('データ取得エラー:', error);
        return null;
    }
}

実用例2: ファイル処理

// 複数のファイルを処理する例
async function processFiles(files) {
    const results = [];
    
    for (const file of files) {
        try {
            console.log(`処理中: ${file.name}`);
            
            // ファイルを読み込む
            const content = await readFile(file);
            
            // 内容を処理
            const processed = await processContent(content);
            
            // 結果を保存
            await saveResult(file.name, processed);
            
            results.push({
                file: file.name,
                status: 'success'
            });
            
        } catch (error) {
            results.push({
                file: file.name,
                status: 'error',
                error: error.message
            });
        }
    }
    
    return results;
}

// ヘルパー関数（模擬）
async function readFile(file) {
    await delay(100);
    return `${file.name}の内容`;
}

async function processContent(content) {
    await delay(200);
    return content.toUpperCase();
}

async function saveResult(filename, content) {
    await delay(100);
    console.log(`保存完了: ${filename}`);
}

// アニメーションを順番に実行
async function runAnimation() {
    const box = document.getElementById('animated-box');
    
    // 右に移動
    await animateBox(box, 'right', 200);
    
    // 下に移動
    await animateBox(box, 'down', 100);
    
    // 左に移動

    await animateBox(box, 'up', 100);
    
    // 完了メッセージ
    box.textContent = '✅';
}

async function animateBox(element, direction, distance) {
    return new Promise(resolve => {
        element.style.transition = 'transform 0.5s ease';
        
        switch(direction) {
            case 'right':
                element.style.transform = `translateX(${distance}px)`;
                break;
            case 'down':
                element.style.transform += ` translateY(${distance}px)`;
                break;
            // ... 他の方向
        }
        
        setTimeout(resolve, 500);
    });
}

ベストプラクティス

1. エラーハンドリングを忘れない

// ❌ 悪い例
async function badExample() {
    const data = await fetch('/api/data'); // エラーが発生したらクラッシュ
}

// ✅ 良い例
async function goodExample() {
    try {
        const data = await fetch('/api/data');
        return data;
    } catch (error) {
        console.error('エラー:', error);
        return null; // デフォルト値を返す
    }
}

2. 不要な await を避ける

// ❌ 不要な await
async function unnecessary() {
    return await fetch('/api/data'); // awaitは不要
}

// ✅ シンプルに
async function simple() {
    return fetch('/api/data');
}

3. 並列処理を活用する

// ❌ 逐次処理（遅い）
const user = await getUser();
const posts = await getPosts();
const comments = await getComments();

// ✅ 並列処理（速い）
const [user, posts, comments] = await Promise.all([
    getUser(),
    getPosts(),
    getComments()
]);

まとめ

async/awaitの重要ポイント

  • ✅ Promiseをより直感的に書ける
  • ✅ 同期的なコードのように読みやすい
  • ✅ try-catchでエラーハンドリングが簡単
  • ✅ デバッグがしやすい
  • ✅ Promiseと完全に互換性がある
